\section{Overview}

PyFBB provides comprehensive support for AX.25 packet radio integration through multiple transport layers:

\begin{itemize}
  \item Direct TCP -- for local or network-based forwarding
  \item AGWPE -- classic Windows AGW Packet Engine TCP/IP interface (port 8000)
  \item KISS/XKISS -- over serial port or TCP
  \item Connected-mode AX.25 -- full link layer with SABM/UA, I-frames, RR polling
\end{itemize}

\section{Prerequisites}

\begin{lstlisting}[language=bash]
pip install pyfbb

\end{lstlisting}

\section{AGWPE Transport}

\begin{lstlisting}[language=python]
from pyfbb import FBBForwarder

fwd = FBBForwarder(
    my_call="KE4AHR",
    remote_call="BBS-1",
    transport_type="agw",
    host="127.0.0.1",
    port=8000
)
fwd.connect(remote_call="BBS-1")
\end{lstlisting}

\section{KISS/XKISS Transport}

\begin{lstlisting}[language=python]
from pyfbb.fbb.transport import KISSTransport, AX25Connection
from pyfbb import FBBForwarder

# Serial example
kiss = KISSTransport(serial_port="/dev/ttyUSB0", baud=9600)
ax25 = AX25Connection(kiss, "KE4AHR", "BBS-1", path=["RELAY1"])
fwd = FBBForwarder(transport=ax25)
fwd.connect()
\end{lstlisting}
